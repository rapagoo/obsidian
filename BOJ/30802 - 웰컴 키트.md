
https://www.acmicpc.net/problem/30802

```cpp
#include <iostream>

int main()
{
	int N{};
	std::cin >> N;
	int arr[6]{};
	for (int i{}; i < 6; ++i)
	{
		std::cin >> arr[i];
	}
	int T{}, P{};
	std::cin >> T >> P;
	int countT{};
	for (int i{}; i < 6; ++i)
	{
		if ((arr[i] / T) != 0)
		{
			countT += (arr[i] + T - 1) / T;
		}
		else if (arr[i] != 0) ++countT;
	}
	std::cout << countT << '\n';
	int countP{ N / P };
	int remainP{ N % P };
	std::cout << countP << ' ' << remainP << '\n';
	return 0;
}
```

`countT += (arr[i] + T - 1) / T;` 이 라인이 핵심
a / b의 소수점 자리 올림 값을 구하려면, (a+b-1)/b 하면 된다.
